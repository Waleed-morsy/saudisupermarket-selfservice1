        <button onclick="search()">عرض البيانات</button>

<script>
function search() {
    const code = document.getElementById("empCode").value.trim();
    const nid = document.getElementById("nid").value.trim();
    const resultDiv = document.getElementById("result");

    resultDiv.innerHTML = "⏳ جاري البحث...";

    fetch("YOUR_FLOW_URL_HERE", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ EmpCode: code, NationalID: nid })
    })
    .then(res => res.json())
    .then(data => {
        if (!data || data.status === "not_found") {
            resultDiv.innerHTML = "<div class='item'>❌ الموظف غير موجود</div>";
            return;
        }

        resultDiv.innerHTML = `
            <div class='item'><strong>الاسم:</strong> ${data.name}</div>
            <div class='item'><strong>الشهر:</strong> ${data.month}</div>
            <div class='item'><strong>إجمالي التأخيرات:</strong> ${data.late}</div>
            <div class='item'><strong>السنوي المتبقي:</strong> ${data.annual}</div>
            <div class='item'><strong>إجمالي الخصم الشهري:</strong> ${data.deduct}</div>
            <div class='item'><strong>أيام الغياب:</strong> ${data.absent}</div>
        `;
    })
    .catch(() => {
        resultDiv.innerHTML = "<div class='item'>⚠️ خطأ في الاتصال بالخدمة</div>";
    });
}
</script>
